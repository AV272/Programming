This file contain base functions of FORTRAN language.

1) gfortan hello.f90 -o hello / -- компиляция программы
2) ./hello / -- запуск программы

3) Data types:
	integer
	real
	complex
	character - text data
	logical
	
4) Variable declaration:
	integer :: amount = 10
	real :: pi = 3.1415
	complex :: frequency = (1.0, -0.5)
	character :: initial = 'A'
	logical :: isOkay = .false. or .true.
	
5) input/output:
	print *, 'comment', variable
	read(*,*) variable
	
6) Expressions:
	** - exponent
	*  - multiplication
	/  - division
	+  - addition
	-  - substraction
